<?php
require_once 'configuraciónInicial.php';
if (session_status() === PHP_SESSION_NONE) {
    session_start();
}

// Redirigir si el usuario no está logueado
if (!isset($_SESSION["user_id"]) || $_SESSION["loggedin"] !== true) {
    header("Location: " . BASE_URL . "bienvenida.php");
    exit();
}

$userId = $_SESSION["user_id"];
$userName = $_SESSION["user_name"] ?? 'Usuario';
$userRoleGlobal = $_SESSION["id_rol"] ?? null;
$userEmpresaIdSession = $_SESSION["id_empresa"] ?? null; // ID de la empresa asociada en sesión
$userRolEmpresaSession = $_SESSION["id_rol_empresa"] ?? null; // Rol dentro de la empresa asociada

$link = getDbConnection();

$selectedCompanyId = null;
$companyInfo = null;
$internalUsers = [];

$mensaje = '';
$tipoMensaje = '';

// Roles de empresa que pueden acceder a esta sección (Admin Empresa, Contratador, Empleado Interno)
$allowedCompanyRoles = [1, 2, 3];

// Obtener el ID del rol 'ADMINISTRADOR' global
$stmtGlobalAdminRole = $link->prepare("SELECT id FROM ROL WHERE nombre_rol = 'ADMINISTRADOR'");
$stmtGlobalAdminRole->execute();
$globalAdminRoleId = $stmtGlobalAdminRole->fetchColumn();
$isGlobalAdmin = ($userRoleGlobal == $globalAdminRoleId);

// Determinar el ID de la empresa a mostrar
if (isset($_GET['id_empresa']) && is_numeric($_GET['id_empresa'])) {
    $selectedCompanyId = (int)$_GET['id_empresa'];
} else {
    // Si no viene en GET, intentar usar el de la sesión para Contratadores/Empleados Internos
    // o redirigir a seleccionar_empresa.php si es un Admin de Empresa sin selección
    if (in_array($userRolEmpresaSession, [2, 3])) {
        $selectedCompanyId = $userEmpresaIdSession;
    } else {
        // Si no es Contratador/Empleado Interno y no hay id_empresa en GET,
        // redirigir a la página de selección de empresa.
        header("Location: " . BASE_URL . "seleccionar_empresa.php");
        exit();
    }
}

// Validar que el usuario tiene permiso para ver la empresa seleccionada
$hasPermissionToViewCompany = false;
try {
    if ($selectedCompanyId) {
        if ($isGlobalAdmin) {
            // El administrador global puede ver cualquier empresa
            $hasPermissionToViewCompany = true;
        } else {
            // Verificar si el usuario está asociado a la empresa con un rol permitido
            $stmtCheckAccess = $link->prepare("
                SELECT COUNT(*) FROM USUARIO_EMPRESA
                WHERE id_usuario = ? AND id_empresa = ? AND id_rol_empresa IN (" . implode(',', $allowedCompanyRoles) . ")
            ");
            $stmtCheckAccess->execute([$userId, $selectedCompanyId]);
            if ($stmtCheckAccess->fetchColumn() > 0) {
                $hasPermissionToViewCompany = true;
            }
        }
    }
} catch (PDOException $e) {
    error_log("Error al verificar permisos de empresa: " . $e->getMessage());
    $mensaje = "Error al verificar permisos de acceso a la empresa.";
    $tipoMensaje = "danger";
}


if (!$hasPermissionToViewCompany) {
    $_SESSION['error'] = "No tienes permiso para acceder al equipo de esta empresa.";
    header("Location: " . BASE_URL . "dashboard.php");
    exit();
}

try {
    // Obtener información de la empresa seleccionada/asociada
    $stmtCompanyInfo = $link->prepare("
        SELECT nombre_empresa, descripcion, email_contacto, telefono_contacto, url_sitio_web, logo_ruta
        FROM EMPRESA
        WHERE id_empresa = ?
    ");
    $stmtCompanyInfo->execute([$selectedCompanyId]);
    $companyInfo = $stmtCompanyInfo->fetch(PDO::FETCH_ASSOC);

    if ($companyInfo) {
        // Procesar ruta del logo
        $companyInfo['logo_url'] = BASE_URL . 'images/default_logo.png'; // Fallback
        if (!empty($companyInfo['logo_ruta'])) {
            $logoFullPath = ROOT_PATH . 'uploads/logos_empresa/' . basename($companyInfo['logo_ruta']);
            $logoPublicPath = BASE_URL . 'uploads/logos_empresa/' . basename($companyInfo['logo_ruta']);
            if (file_exists($logoFullPath)) {
                $companyInfo['logo_url'] = $logoPublicPath;
            }
        }

        // Obtener usuarios internos de la empresa
        $stmtInternalUsers = $link->prepare("
            SELECT U.id, U.nombre, U.email, U.foto_perfil, RE.nombre_rol_empresa
            FROM USUARIO_EMPRESA UE
            JOIN USUARIO U ON UE.id_usuario = U.id
            JOIN ROL_EMPRESA RE ON UE.id_rol_empresa = RE.id_rol_empresa
            WHERE UE.id_empresa = ?
            ORDER BY U.nombre ASC
        ");
        $stmtInternalUsers->execute([$selectedCompanyId]);
        $internalUsers = $stmtInternalUsers->fetchAll(PDO::FETCH_ASSOC);

        // Procesar fotos de perfil de usuarios internos
        foreach ($internalUsers as &$user) {
            $user['profile_image_url'] = BASE_URL . 'images/default-profile.jpg';
            if (!empty($user['foto_perfil'])) {
                $profileFullPath = ROOT_PATH . 'uploads/profile_pictures/' . basename($user['foto_perfil']);
                $profilePublicPath = BASE_URL . 'uploads/profile_pictures/' . basename($user['foto_perfil']);
                if (file_exists($profileFullPath)) {
                    $user['profile_image_url'] = $profilePublicPath;
                }
            }
        }
        unset($user); // Romper la referencia

    } else {
        $mensaje = "No se encontró información para la empresa seleccionada.";
        $tipoMensaje = "danger";
    }

    // --- Lógica para crear chat grupal ---
    if ($_SERVER["REQUEST_METHOD"] === "POST" && isset($_POST['create_group_chat'])) {
        $selectedUserIds = $_POST['selected_users'] ?? [];
        $groupChatName = trim($_POST['group_chat_name'] ?? '');

        if (empty($selectedUserIds)) {
            $mensaje = "Debes seleccionar al menos un usuario para crear un chat grupal.";
            $tipoMensaje = "danger";
        } else {
            // Asegurarse de que el usuario actual también esté en el chat si no está seleccionado
            if (!in_array($userId, $selectedUserIds)) {
                $selectedUserIds[] = $userId;
            }

            // Validar que los usuarios seleccionados pertenecen a la empresa actual
            $validUserIds = [];
            foreach ($internalUsers as $u) {
                if (in_array($u['id'], $selectedUserIds)) {
                    $validUserIds[] = $u['id'];
                }
            }

            if (count($validUserIds) < 2) { // Un chat grupal necesita al menos 2 participantes (incluyendo el creador)
                $mensaje = "Selecciona al menos otro usuario además de ti para crear un chat grupal.";
                $tipoMensaje = "danger";
            } else {
                $link->beginTransaction();
                try {
                    // Generar un título si no se proporciona uno
                    if (empty($groupChatName)) {
                        $names = [];
                        foreach ($validUserIds as $id) {
                            foreach ($internalUsers as $u) {
                                if ($u['id'] == $id) {
                                    $names[] = $u['nombre'];
                                    break;
                                }
                            }
                        }
                        $groupChatName = "Chat de equipo: " . implode(", ", array_slice($names, 0, 3)) . (count($names) > 3 ? "..." : "");
                    }

                    // Insertar nueva conversación
                    $stmtInsertConv = $link->prepare("
                        INSERT INTO CONVERSACION (tipo_conversacion, titulo_conversacion, fecha_creacion, id_proyecto)
                        VALUES ('empresa_interna', ?, NOW(), ?) -- CORREGIDO: 'equipo_interno' a 'empresa_interna'
                    ");
                    // id_proyecto se puede usar para almacenar el id_empresa si es un chat de equipo
                    $stmtInsertConv->execute([$groupChatName, $selectedCompanyId]);
                    $newChatId = $link->lastInsertId();

                    // Añadir participantes a la conversación
                    $stmtInsertParticipant = $link->prepare("
                        INSERT INTO CONVERSACION_PARTICIPANTE (id_conversacion, id_usuario)
                        VALUES (?, ?)
                    ");
                    foreach ($validUserIds as $participantId) {
                        $stmtInsertParticipant->execute([$newChatId, $participantId]);
                    }

                    $link->commit();
                    $_SESSION['mensaje'] = "Chat grupal creado exitosamente.";
                    $_SESSION['tipo'] = "success";
                    header("Location: " . BASE_URL . "mis_chats.php?chat_id=" . $newChatId);
                    exit();

                } catch (PDOException $e) {
                    $link->rollBack();
                    error_log("Error al crear chat grupal: " . $e->getMessage());
                    $mensaje = "Error al crear el chat grupal: " . $e->getMessage();
                    $tipoMensaje = "danger";
                }
            }
        }
    }

} catch (PDOException $e) {
    error_log("Error PDO en mi_equipo.php: " . $e->getMessage());
    $mensaje = "Error de base de datos al cargar la página.";
    $tipoMensaje = "danger";
} catch (Exception $e) {
    error_log("Error general en mi_equipo.php: " . $e->getMessage());
    $mensaje = "Ocurrió un error inesperado.";
    $tipoMensaje = "danger";
}

            <?= htmlspecialchars($mensaje) ?>
            <button type="button" class="btn-close" data-bs-dismiss="alert" aria-label="Close"></button>
        </div>
    <?php endif; ?>

    <h2 class="mb-4 text-center">

    <?php if ($companyInfo): ?>
        <div class="card company-info-card mb-4">

                <h3><?= htmlspecialchars($companyInfo['nombre_empresa']) ?></h3>
                <p class="text-muted"><?= htmlspecialchars($companyInfo['descripcion'] ?? 'Sin descripción.') ?></p>
                <p><i class="fas fa-envelope me-2"></i><?= htmlspecialchars($companyInfo['email_contacto']) ?></p>
                <p><i class="fas fa-phone me-2"></i><?= htmlspecialchars($companyInfo['telefono_contacto']) ?></p>
                <?php if (!empty($companyInfo['url_sitio_web'])): ?>
                    <p><i class="fas fa-globe me-2"></i><a href="<?= htmlspecialchars($companyInfo['url_sitio_web']) ?>" target="_blank"><?= htmlspecialchars($companyInfo['url_sitio_web']) ?></a></p>
                <?php endif; ?>
            </div>
        </div>

        <h3 class="mb-3">Miembros del Equipo</h3>

        <?php if (empty($internalUsers)): ?>
            <div class="alert alert-info text-center">No hay miembros en el equipo de esta empresa.</div>
        <?php else: ?>
            <form id="groupChatForm" method="POST">
                <input type="hidden" name="create_group_chat" value="1">
                
                <div class="d-flex justify-content-end mb-3">

                    </button>

                <div class="row">
                    <?php foreach ($internalUsers as $user): ?>

                                    <p class="card-text">
                                        <span class="badge bg-secondary"><?= htmlspecialchars($user['nombre_rol_empresa']) ?></span>
                                    </p>
                                    <div class="d-flex justify-content-center gap-2 mt-3">
                                        <?php if ($user['id'] != $userId): // No mostrar chat privado consigo mismo ?>

                                                <i class="fas fa-comment"></i> Chat Privado

                                            <label class="form-check-label" for="user_<?= htmlspecialchars($user['id']) ?>">

<script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/js/bootstrap.bundle.min.js"></script>
<script>
document.addEventListener('DOMContentLoaded', function() {
    const createGroupChatModal = document.getElementById('createGroupChatModal');
    const selectedUsersList = document.getElementById('selectedUsersList');
    const userCheckboxes = document.querySelectorAll('.user-checkbox');
    const confirmCreateGroupChatBtn = document.getElementById('confirmCreateGroupChatBtn');
    const groupChatForm = document.getElementById('groupChatForm');

    // Función para actualizar la lista de usuarios seleccionados en el modal
    function updateSelectedUsersInModal() {
        selectedUsersList.innerHTML = '';
        let selectedCount = 0;
        const currentUserId = <?= json_encode($userId) ?>; // Obtener el ID del usuario actual de PHP

        userCheckboxes.forEach(checkbox => {
            const userId = checkbox.value;
            const userName = checkbox.closest('.user-card').querySelector('.card-title').textContent;

            // Asegurarse de que el usuario actual siempre esté en la lista del modal, incluso si su checkbox está deshabilitado
            if (checkbox.checked || userId == currentUserId) {
                const listItem = document.createElement('li');
                listItem.classList.add('list-group-item');
                listItem.textContent = userName + ((userId == currentUserId) ? ' (Tú)' : '');
                selectedUsersList.appendChild(listItem);
                selectedCount++;
            }
        });

        // Deshabilitar el botón de crear chat si no hay al menos 2 usuarios seleccionados (incluyendo el actual)
        if (selectedCount < 2) {
            confirmCreateGroupChatBtn.disabled = true;
            const warningItem = document.createElement('li');
            warningItem.classList.add('list-group-item', 'list-group-item-warning');
            warningItem.textContent = 'Selecciona al menos otro usuario para crear un chat grupal.';
            selectedUsersList.appendChild(warningItem);
        } else {
            confirmCreateGroupChatBtn.disabled = false;
        }
    }

    // Event listener para abrir el modal
    if (createGroupChatModal) {
        createGroupChatModal.addEventListener('show.bs.modal', function() {
            updateSelectedUsersInModal();
        });
    }

    // Event listener para cambios en los checkboxes
    userCheckboxes.forEach(checkbox => {
        checkbox.addEventListener('change', function() {
            // No es necesario actualizar el modal aquí, solo cuando se abre
            // La función updateSelectedUsersInModal se encarga de eso.
        });
    });

    // Asegurarse de que el checkbox del usuario actual esté marcado y deshabilitado al cargar
    userCheckboxes.forEach(checkbox => {
        const userId = checkbox.value;
        const currentUserId = <?= json_encode($userId) ?>;
        if (userId == currentUserId) {
            checkbox.checked = true;
            checkbox.disabled = true;
        }
    });
});
</script>
